function switchLang() {
  const currentDir = document.body.getAttribute("dir");
  const newLang = currentDir === "rtl" ? "en" : "fa";
  setLang(newLang);

 
  localStorage.setItem("siteLang", newLang);
}

function setLang(lang) {
  const faTexts = document.querySelectorAll(".lang.fa");
  const enTexts = document.querySelectorAll(".lang.en");

  if (lang === "en") {
    document.body.setAttribute("dir", "ltr");
    faTexts.forEach(el => el.style.display = "none");
    enTexts.forEach(el => el.style.display = "block");
  } else {
    document.body.setAttribute("dir", "rtl");
    faTexts.forEach(el => el.style.display = "block");
    enTexts.forEach(el => el.style.display = "none");
  }
}
